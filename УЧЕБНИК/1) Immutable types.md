    z = []

    if len(a) < len(b):
        a, b = b, a

    carry = 0

    for i in range(0, len(b)):
        carry += a[i] + b[i]
        z.append(carry % MASK)
        carry = carry // MASK

    for i in range(i + 1, len(a)):
        carry += a[i]
        z.append(carry % MASK)
        carry = carry // MASK

    z.append(carry)

    # –£–¥–∞–ª–∏–º –∑–∞–≤–µ—Ä—à–∞—é—â–∏–µ –Ω—É–ª–∏
    i = len(z)
    while i > 0 and z[i-1] == 0:

    z = z[0:i]

    return z
